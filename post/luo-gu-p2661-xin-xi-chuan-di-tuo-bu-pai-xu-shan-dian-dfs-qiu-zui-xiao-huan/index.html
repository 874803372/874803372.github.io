#include<iostream>
#include<cstdio>

using namespace std;

const int N = 2e5 + 10;

int a[N];
int d[N];
int vis[N];
int n;
int ans = 1e9;

void clear(int i) {
	vis[i] = 1;
	d[a[i]]--;
	if(d[a[i]] == 0 && vis[a[i]] != 1) {
		clear(a[i]);
	}
}

void dfs(int x,int l,int cnt) {
	if(x == l && cnt > 0) {
		ans = min(ans, cnt);
		return;
	}
	if(vis[a[x]] == 0) {
		vis[a[x]] = 1;
		dfs(a[x],l,cnt + 1);
	}
}

int main() {
	cin >> n;
	for(int i = 1;i <= n;i++) {
		scanf("%d", &a[i]);
		d[a[i]]++;
	}
	for(int i = 1;i <= n;i++) {
		if(d[i] == 0 && vis[i] != 1) {
			clear(i);
		}
	}
	for(int i = 1;i <= n;i++) {
		if(d[i]) {
			dfs(i,i,0);
		}
	}
	cout << ans << endl;
	return 0;
}
