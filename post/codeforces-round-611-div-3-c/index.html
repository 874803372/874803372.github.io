Codeforces Round #611 (Div. 3) C

https://874803372.github.io/post/codeforces-round-611-div-3-c/
https://874803372.github.io

热度🔥: loading...
  • 题意
    有n个人，n个人之间要互相送礼物，一个人最多只能送出一个礼物和收一个礼物。先输入多少人，然后第二行输入要送给第几个人，如果还没决定送给第几个人就输入0同时一个人不能送礼物给自己。
  • 样例解释
    不是多组输入
    Input
    5
    5 0 0 2 4
    7
    7 4 0 3 0 5 1
    5
    2 1 0 0 0
    output
    5 3 1 2 4
    7 4 2 3 6 5 1
    2 1 4 5 3
    对于第一个样例：第一个人送给了第五个，第四个人送给了第二个，第五个人送给了第四个，还剩第一个和第三个人没有收到礼物，第三个人送礼物不能送给自己所以只有5 3 1 2 4一个方案
  • 思路
    比较典型的构造题，其实样例不止上面三个，但这三个基本就可以代表这个题所有的类型不得不让人感叹codeforces样例真的是对于调试程序很友好了不像某牛。
    我们会发现只有没送礼物也没收到礼物的人才有可能会送礼物给自己。第一个样例，第三个人没有收到礼物同时也没有决定送谁礼物，第二个人还没决定送谁礼物但已经收到礼物，所以第二个人必须送给第三个人礼物。
    第一种情况：没送礼物也没收礼物的，和已经收到礼物但没送礼物的都有
    第二种情况：所有人都已经收到礼物或者送出礼物。
    第三种情况：所有没送出礼物的人也没收到礼物
    针对这三种情况进行构造即可
#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

const int N = 2e5 + 10;

struct Node {
    int id;
    bool vis;
}a[N];

vector<int> last;
vector<int> v;

int main() {
    int n;
    cin >> n;
    for(int i = 1;i <= n;i++) {
        int x;
        scanf("%d", &x);
        a[i].id = x;
        a[x].vis = true;
    }
    for(int i = 1;i <= n;i++) {
        if(a[i].vis == 0) {
            if(a[i].id == 0) {
                v.push_back(i);
            }
            else {
                last.push_back(i);
            }
        }
    }
    int len1 = last.size();
    if(len1 == 0) {
        int len = v.size();
        int temp = 1;
        for(int i = 1;i <= n;i++) {
            if(a[i].id == 0) {
                if(temp == len) {
                    a[i].id = v[0];
                }
                else {
                    a[i].id = v[temp];
                    temp++;
                }
            }
            printf("%d ",a[i].id);
        }
    }
    else {
        int len = v.size();
        for(int i = 0;i < len;i++) {
            a[v[i]].id = last[len1 - 1];
            last.pop_back();
            last.push_back(v[i]);
        }
        int len2 = last.size();
        for(int i = 1;i <= n;i++) {
            if(a[i].id == 0) {
                a[i].id = last[len2 - 1];
                last.pop_back();
                len2--;
            }
            printf("%d ", a[i].id);
        }
    }
    return 0;
}
